44  med_mesh_type _meshtype;

48  med_axis_type _axistype=MED_UNDEF_AXIS_TYPE;
49  /*
50  * On inhibe le gestionnaire d'erreur
51  */
52  _MEDmodeErreurVerrouiller();

54 
55  /*
56  * On regarde si le groupe existe => erreur si non
57  */
58  strcat(_path,meshname);
59  if ((_meshid = _MEDdatagroupOuvrir(fid,_path)) < 0) {
60  MED_ERR_(_ret,MED_ERR_OPEN,MED_ERR_DATAGROUP,_path);
61  ISCRUTE_id(_meshid);goto ERROR;
62  }
63 
64  /*
65  * Si le maillage est de type MED_UNSTRUCTURED_MESH => erreur
66  */
67  if ( _MEDattrEntierLire(_meshid,MED_NOM_TYP,&_intmeshtype) < 0) {
68  MED_ERR_(_ret,MED_ERR_READ,MED_ERR_ATTRIBUTE,MED_ERR_MESH_MSG);
69  SSCRUTE(meshname);SSCRUTE(MED_NOM_TYP); goto ERROR;
70  }
71  _meshtype = (med_mesh_type) (_intmeshtype);
72 
73  if (_meshtype == MED_UNSTRUCTURED_MESH) {
74  MED_ERR_(_ret,MED_ERR_INVALID,MED_ERR_ATTRIBUTE,MED_ERR_MESH_MSG);
75  SSCRUTE(meshname);SSCRUTE(MED_NOM_TYP);ISCRUTE_int(_meshtype); goto ERROR;
76  }
77 
78  /* - Une grille MED_CARTESIAN_GRID doit utiliser un système de coordonnées
79  * MED_CARTESIAN
80  * - Une grille MED_POLAR_GRID peut utiliser un système de coordonnées
81  * MED_CYLINDRICAL ou MED_SPHERICAL
82  * - Une Grille MED_CURVILINEAR_GRID peut utiliser un système de coordonnées
83  * quelconque.
84  */
85  if ( _MEDattrEntierLire(_meshid,MED_NOM_REP,&_intaxistype) < 0) {
86  MED_ERR_(_ret,MED_ERR_READ,MED_ERR_ATTRIBUTE,MED_ERR_MESH_MSG);
87  SSCRUTE(meshname);SSCRUTE(MED_NOM_REP); goto ERROR;
88  }
89  _axistype = (med_axis_type) (_intaxistype);
90 
91  if (
92  ( (gridtype == MED_CARTESIAN_GRID) && (_axistype != MED_CARTESIAN ) ) ||
93  ( (gridtype == MED_POLAR_GRID ) && ((_axistype != MED_CYLINDRICAL) &&
94  (_axistype != MED_SPHERICAL ) ))
95  ) {
96  MED_ERR_(_ret,MED_ERR_RANGE,MED_ERR_AXISTYPE,MED_ERR_MESH_MSG);
97  SSCRUTE(meshname);ISCRUTE(_intaxistype);goto ERROR;
98  }
99 
100 
101  /*
102  * Creation de l'attribut correspondant au type de grille
103  * L'attribut "GTY"
104  */
105  if ( _MEDattributeIntWr(_meshid,MED_NOM_GTY,&_intgridtype) < 0) {
106  MED_ERR_(_ret,MED_ERR_WRITE,MED_ERR_ATTRIBUTE,MED_ERR_MESH_MSG);
107  SSCRUTE(meshname);SSCRUTE(MED_NOM_GTY);
108  ISCRUTE(_intgridtype);goto ERROR;
109  }
110 
111  _ret = 0;
112

114 
115  if (_meshid>0) if (_MEDdatagroupFermer(_meshid) < 0) {
116  MED_ERR_(_ret,MED_ERR_CLOSE,MED_ERR_DATAGROUP,_path);
117  ISCRUTE_id(_meshid);
118  }
119 
120  return _ret;
121 }
