40  5000., 6000., 7000., 8000.};

42  const med_float tria3values_step2[8] = {1500., 2500., 3500., 4500.,
43  5500., 6500., 7500., 8500.};
44  const med_float quad4values_step2[4] = {15000., 25000., 35000., 45000.};
45  int ret=-1;
46 
47  /* file creation */
48  fid = MEDfileOpen("UsesCase_MEDfield_4.med",MED_ACC_CREAT);
49  if (fid < 0) {

51  goto ERROR;

53 
54  /* create mesh link */

56  MESSAGE("ERROR : create mesh link ...");
57  goto ERROR;
58  }
59 
60  /* field creation : temperature field : 1 component in celsius degree
61  * the mesh is the 2D unstructured mesh of UsescaseMEDmesh_1.c
62  * use case. Computation step unit in 'ms'
63  */
64  if (MEDfieldCr(fid, fieldname, MED_FLOAT64,
65  ncomponent, componentname, componentunit,"ms", meshname) < 0) {
66  MESSAGE("ERROR : create field");
67  goto ERROR;
68  }

71  * - first on meshname MED_NO_DT,MED_NO_IT mesh computation step

73  /* write values at cell centers : 8 MED_TRIA3 and 4 MED_QUAD4 */
74 
75  /* STEP 1 : dt1 = 5.5, it = 1*/
76  /* MED_TRIA3 */
77  if (MEDfieldValueWr(fid, fieldname, 1, 1, 5.5, MED_CELL, MED_TRIA3, MED_FULL_INTERLACE,
78  MED_ALL_CONSTITUENT, ntria3, (unsigned char*) tria3values_step1) < 0) {
79  MESSAGE("ERROR : write field values on MED_TRIA3");
80  goto ERROR;
81  }
82  /* MED_QUAD4 */
83  if (MEDfieldValueWr(fid, fieldname, 1, 1, 5.5, MED_CELL,MED_QUAD4, MED_FULL_INTERLACE,
84  MED_ALL_CONSTITUENT, nquad4, (unsigned char*) quad4values_step1) < 0) {
85  MESSAGE("ERROR : write field values on MED_QUAD4 ");
86  goto ERROR;
87  }
88 
89  /* STEP 2 : dt2 = 8.9, it = 1*/
90  /* MED_TRIA3 */
91  if (MEDfieldValueWr(fid, fieldname, 2 , 1 , 8.9 , MED_CELL, MED_TRIA3, MED_FULL_INTERLACE,
92  MED_ALL_CONSTITUENT, ntria3, (unsigned char*)tria3values_step2) < 0) {
93  MESSAGE("ERROR : write field values on MED_TRIA3");
94  goto ERROR;
95  }
96 
97  /* MED_QUAD4 */
98  if (MEDfieldValueWr(fid, fieldname, 2, 1, 8.9, MED_CELL, MED_QUAD4, MED_FULL_INTERLACE,
99  MED_ALL_CONSTITUENT, nquad4, (unsigned char*)quad4values_step2) < 0) {
100  MESSAGE("ERROR : write field values on MED_QUAD4 ");
101  goto ERROR;
102  }
103 
104  /* Write associated mesh computation step */
105  if ( MEDfieldComputingStepMeshWr(fid, fieldname, 2, 1,
106  1, 3 ) < 0 ) {
107  MESSAGE("ERROR : write field mesh computation step error ");
108  goto ERROR;
109  }
110 
111  ret=0;
112  ERROR:
113 
114  /* close file */
115  if (MEDfileClose(fid) < 0) {
116  MESSAGE("ERROR : close file ...");
117  ret=-1;
118  }
119 
120  return ret;
121 }
